58 void i2cSetBitrate(uint16_t bitratekHz)
59 {
60  uint8_t bitrate_div;
61  // set i2c bitrate
62  // Set Prescaler to one --> TWPS = 1
63  TWSR =((0<<TWPS0)|(0<<TWPS1));
64  //calculate bitrate division
65  bitrate_div = (((float)F_CPU/1000.0)/(float)bitratekHz);
66  if(bitrate_div >= 16){
67  bitrate_div = (float)(bitrate_div-16)/8;
68  }
69  //printf("DIV: %d \n",bitrate_div);
70  TWBR=(uint8_t)bitrate_div;
71  //TWBR=0; -> max speed
72  //PORTC |= ((1<<PINC4) | (1<<PINC5));
73 }
